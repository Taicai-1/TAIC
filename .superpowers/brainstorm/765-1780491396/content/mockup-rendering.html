def audit_financier(contrat):
    """Évalue la viabilité financière d'un contrat."""
    marge = contrat.revenu * 0.35
    cout_mensuel = 60_000  # infra + support

    if contrat.tresorerie < 3_500_000:
        return {"decision": "REFUSER", "raison": "Trésorerie insuffisante"}

    roi = (marge - cout_mensuel) * 12
    return {"decision": "ACCEPTER", "roi_annuel": roi}

Décision finale : accepter sous conditions — acompte de 30%, délai de paiement réduit à 60 jours, et ligne de crédit sécurisée de 4M€.
